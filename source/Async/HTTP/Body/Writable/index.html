def initialize(length = nil, queue: Async::Queue.new)
	@queue = queue
	
	@length = length
	
	@count = 0
	
	@finished = false
	
	@closed = false
	@error = nil
end

def close(error = nil)

Stop generating output; cause the next call to write to fail with the given error.

Implementation

 
def close(error = nil)
	unless @closed
		@queue.enqueue(nil)
		
		@closed = true
		@error = error
	end
	
	super
end

def read
	return if @finished
	
	unless chunk = @queue.dequeue
		@finished = true
	end
	
	return chunk
end

def write(chunk)

Write a single chunk to the body. Signal completion by calling #finish.

Implementation

 
def write(chunk)
	# If the reader breaks, the writer will break.
	# The inverse of this is less obvious (*)
	if @closed
		raise(@error || Closed)
	end
	
	@count += 1
	@queue.enqueue(chunk)
end
